//js
var menu = {
  nasi:0,
  addon: 0,
  tomyum:0
}
document.getElementById('addNasi').addEventListener("click", addNasi);
document.getElementById('addTomyum').addEventListener("click", addTomyum);
document.getElementById('placeOrder').addEventListener("click", placeOrder);
function addNasi() {
  menu.nasi += document.getElementById('nasi-quantity').value;
  menu.addon += document.getElementById('nasi-addon').value;
  document.getElementById('nasi-quantity').value = 0;
  document.getElementById('nasi-addon').value = 0;
  if (menu.nasi != 0) {
    document.getElementById("left-nasi").innerHTML = "Nasi goreng kampung ala Thai x"+menu.nasi;
    document.getElementById("right-nasi").innerHTML = Number(menu.nasi * 10).toFixed(2);
  }
  if (menu.addon != 0) {
    document.getElementById("left-addon").innerHTML = "Add-On Hard Boiled Egg x"+menu.addon;
    document.getElementById("right-addon").innerHTML = Number(menu.addon * 1).toFixed(2);
  }
}

function addTomyum() {
  menu.tomyum += document.getElementById('tomyum-quantity').value;
  document.getElementById('tomyum-quantity').value = 0;
  if (menu.tomyum != 0) {
    document.getElementById("left-tomyum").innerHTML = "Prawn tom yum soup x"+menu.tomyum;
    document.getElementById("right-tomyum").innerHTML = Number(menu.tomyum * 15).toFixed(2);
  }
}

function placeOrder() {
  var total = Number((menu.nasi * 10) + (menu.addon * 1) + (menu.tomyum * 15));
  var sst = Number(total*0.06);
  var grand = Number(sst+total);

  if (total!=0) {
    document.getElementById("left-total").innerHTML = "Total";
    document.getElementById("right-total").innerHTML = total.toFixed(2);
    document.getElementById("left-SST").innerHTML = "SST (6%)";
    document.getElementById("right-SST").innerHTML = sst.toFixed(2);
    document.getElementById("left-grand").innerHTML = "GRAND TOTAL";
    document.getElementById("right-grand").innerHTML = grand.toFixed(2);
  }
}
